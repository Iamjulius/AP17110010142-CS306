#include <stdio.h>

int main(void) {
  int state=0;
  char inp[1000];
  scanf("%s",inp);
  int i=0;
  while(inp[i]){
	    if(state==0){
		    if(inp[i]=='0'){
			  state=1;
		    }
		    else if(inp[i]=='1'){
			  state=3;
		    }
			else{
				printf("Invalid input, String ain't accepted.");
				exit(0);
			}
	    }
		else if(state==1){
		  if(inp[i]=='0'){
			  state=0;
		  }
		  else if(inp[i]=='1'){
			  state=2;
		  }
		  else{
				printf("Invalid input, String ain't accepted.");
				exit(0);
			}
	    }
		else if(state==2){
		  if(inp[i]=='0'){
			  state=3;
		  }
		  else if(inp[i]=='1'){
			  state=1;
		  }
		  else{
				printf("Invalid input, String ain't accepted.");
				exit(0);
			}
	    }
		else if(state==3){
		  if(inp[i]=='0'){
			  state=2;
		  }
		  else if(inp[i]=='1'){
			  state=0;
		  }
		  else{
				printf("Invalid input, String ain't accepted.");
				exit(0);
			}
	    }
	i=i+1;
  }
	if(state==1 || state==2 || state==3) printf("String accepted.");
	else printf("String not accepted!");
  return 0;
}
